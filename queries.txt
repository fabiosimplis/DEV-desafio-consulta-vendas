
SELECT sa.id, sa.date, sa.amount, se.name
FROM  TB_SELLER AS se
INNER JOIN TB_SALES AS sa
    ON se.id = sa.seller_id
WHERE  UPPER(se.name) LIKE CONCAT('%', UPPER('odinson'),'%')
AND date BETWEEN '2022-05-01' AND '2022-05-31'
AND sa.id IN (59, 60, 67, 69, 73)

SELECT sa.id, sa.date, sa.amount, se.name
 FROM TB_SELLER AS se
INNER JOIN TB_SALES AS sa
    ON se.id = sa.seller_id
 WHERE  date
    BETWEEN '2023-04-05'
        AND '2024-04-05'
